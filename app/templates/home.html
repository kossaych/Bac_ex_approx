    def suitex(M,n,k) :
        if k == 0 :
            return M/2
        else :
            Tp = suitex(M,n,k-1)
            return (1 / n) * ((n - 1) * Tp + (M / (Tp ** (n - 1))))

            k = k + 1
            T = suitex(M,n,k)  
        return T
